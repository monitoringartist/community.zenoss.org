    RE: LPAR
    I created a script a while back that would run through "most" of the AIX snmp config needed to get it working...I had mixed success, though.

    Specifically, i've had to run the script twice, because the 'kill -2 xmtopas' step of the process never would create the /etc/perf/xmservd.mib file properly. Running the script a second time after manually executing the 'kill -2 xmtopas' would work though.

    The script automates most of what I've found here and in the Cacti forums, though, and hopefully is fixable by those who want to.

    Please forgive the ugliness of both the code and my ignorance of AIX...we have several AIX boxes in our network (5.1, 5.2, and 5.3), so hopefully this will at least help some folks get their agents configured "mostly" correctly.

    I'd be careful using this code in a production environment (hope that's obvious from the tone here).


    Script follows:

    #!/usr/bin/ksh
    ###################################
    #
    # AIX_snmp.sh
    #
    # Script to set up snmp properly
    # on an AIX box
    #
    # !!!ATTN: This Script should
    # NEVER be run on HACMP
    # Cluster nodes!!!
    #
    ###################################

    PATH=$PATH
    THISDIR=$(pwd)
    DATE=$(date +"%m%d%Y_%H%M")
    # files to change
    INITTAB=/etc/inittab
    INETDCONF=/etc/inetd.conf
    SNMPDPEERS=/etc/snmpd.peers
    SNMPDCONF=/etc/snmpdv3.conf
    MIBDEFS=/etc/mib.defs
    XMSERVDMIB=/etc/perf/xmservd.mib

    # TEST files to change
    #INITTAB=/tmp/snmptest/etc/inittab
    #INETDCONF=/tmp/snmptest/etc/inetd.conf
    #SNMPDPEERS=/tmp/snmptest/etc/snmpd.peers
    #SNMPDCONF=/tmp/snmptest/etc/snmpdv3.conf
    #MIBDEFS=/tmp/snmptest/etc/mib.defs
    #XMSERVDMIB=/tmp/snmptest/etc/perf/xmservd.mib



    ###################################
    # Name: Main
    # Purpose: Execute Code
    ###################################
    Main() {
    echo "Beginning SNMP configuration"
    # verify OS version
    osVersion;

    #topasPrep;


    }


    ###################################
    # Name: osVersion
    # Purpose: version 5.2 config
    ###################################
    osVersion() {
    VERSION=$(oslevel)
    echo "Checking AIX Version"
    #echo $VERSION
    if [ "$(echo ${VERSION} | grep "5.3")" ] ; then
    echo $VERSION
    config53;
    fi

    if [ "$(echo ${VERSION} | grep "5.2")" ] ; then
    echo $VERSION
    config52;
    fi

    if [ "$(echo ${VERSION} | grep "5.1")" ] ; then
    echo $VERSION
    config52;
    fi

    }

    ###################################
    # Name: config53
    # Purpose: version 5.3 config
    ###################################
    config53() {

    # backup/check/change /etc/inittab for xmdaily/topasrec
    fileBackup $INITTAB;
    initTab;

    # reload inittab (init -q)
    echo "Reloading inittab"
    init q

    # backup/check/change /etc/inetd.conf
    fileBackup $INETDCONF;
    inetdConfChange $INETDCONF;

    # refresh inetd
    echo "Refreshing inetd"
    refresh -s inetd

    # backup/edit /etc/snmpd.peers
    fileBackup $SNMPDPEERS;
    snmpdPeers $SNMPDPEERS;

    # backup/edit snmpdv3.conf
    fileBackup $SNMPDCONF;
    snmpdConf $SNMPDCONF;

    # create xmservd.res file (xmtopas <-> smux)
    prepXmservd;

    # stop snmp agents
    snmpdStop;

    # start snmp agents
    snmpdStart;

    # start topas monitor
    topasStart;

    # backup /etc/mib.defs
    fileBackup $MIBDEFS;

    # create xmservd.res file (xmtopas <-> smux)
    prepXmservd;

    # find/kill procs related to topas and xmtopas
    killProc "topas";

    # restart topas
    topasStart;


    mibCompile; # uses crested xmservd.mib file

    # stop snmp agents
    snmpdStop;

    # start snmp agents
    snmpdStart;

    # find/kill procs related to topas and xmtopas
    killProc "topas";

    # restart topas
    topasStart;
    }

    ###################################
    # Name: config53
    # Purpose: start snmp daemons
    ###################################
    config52() {

    # backup/edit /etc/snmpd.peers
    fileBackup $SNMPDPEERS;
    snmpdPeers $SNMPDPEERS;

    # backup/edit snmpdv3.conf
    fileBackup $SNMPDCONF;
    snmpdConf $SNMPDCONF;

    # create xmservd.res file (xmtopas <-> smux)
    #prepXmservd;

    # stop snmp agents
    snmpdStop;

    # start snmp agents
    snmpdStart;

    }

    ###################################
    # Name: fileBackup
    # Purpose: start snmp daemons
    ###################################
    fileBackup() {
    FILE=$1
    BACKUPDIR=$THISDIR/backup
    THISFILE=$(basename $FILE)
    NEWFILE=$BACKUPDIR/$THISFILE.$DATE

    if [[ ! -d $BACKUPDIR ]] ; then
    mkdir -p $BACKUPDIR
    fi

    if [[ ! -e $ARCHIVE ]] ; then
    cp $FILE $NEWFILE
    echo "backing up $FILE to $NEWFILE"
    fi
    }


    ###################################
    # Name: prepXmservd
    # Purpose: start snmp daemons
    ###################################
    prepXmservd() {
    echo "Creating /usr/lpp/perfagent/xmservd.res"
    if [[ ! -d /usr/lpp/perfagent ]] ; then
    mkdir -p /usr/lpp/perfagent
    fi
    touch /usr/lpp/perfagent/xmservd.res
    echo "dosmux" > /usr/lpp/perfagent/xmservd.res
    }


    ###################################
    # Name: topasStart
    # Purpose: start topas daemons
    ###################################
    topasStart() {
    echo "Starting topas"
    nohup topas -C &
    }

    ###################################
    # Name: topasPrep
    # Purpose: start topas daemons
    ###################################
    topasPrep() {
    if [[ ! -d /etc/perf ]] ; then
    mkdir -p /etc/perf
    fi

    PROCS=$(ps -ef | grep xmtopas | grep -v 'grep' | awk '{print $2}')
    PROCS2=$(ps -ef | grep xmtopas | grep -v 'grep')
    print $PROCS
    print $PROCS2
    for PROC in $PROCS ; do
    echo "Sending signal 2 to $PROC"
    kill -2 $PROC
    done
    }

    ###################################
    # Name: KillProc
    # Purpose: kill ID
    ###################################
    killProc() {
    PROCNAME=$1
    PROCS=$(ps -ef | grep $PROCNAME | grep -v 'grep' | awk '{print $2}')
    for PROC in $PROCS ; do
    echo "Sending kill signal to $PROC"
    kill $PROC
    done
    }


    ###################################
    # Name: mibAppend
    # Purpose: append mib.defs
    ###################################
    mibAppend() {
    MIB=$1
    NEWMIB=$2
    echo "Appending mib definitions from $NEWMIB to $MIB"
    cat $NEWMIB >> $MIB
    }


    ###################################
    # Name: mibFix
    # Purpose: append mib.defs
    ###################################
    mibFix() {
    FILE=$1
    echo "Checking $FILE"
    sed -e 's/xmdNFSV3ClntReaddir+/xmdNFSV3ClntReaddirPlus/g' $FILE >>$FILE.tmp
    sed -e 's/xmdNFSV3SvrReaddir+/xmdNFSV3SvrReaddirPlus/g' $FILE.tmp >> $FILE.tmp2
    rm $FILE.tmp
    mv $FILE.tmp2 $FILE
    }


    ###################################
    # Name: mibCompile
    # Purpose: compile new MIB file
    ###################################
    mibCompile() {
    CATMIB=$THISDIR/mib.defs.out

    topasPrep;

    # edit /etc/perf/xmservd.mib#
    fileBackup $XMSERVDMIB;#
    mibFix $XMSERVDMIB;

    echo "Compiling new MIB file"
    mosy -o $CATMIB $XMSERVDMIB
    mibAppend $MIBDEFS $CATMIB;
    }

    ###################################
    # Name: mibCompile
    # Purpose: compile new MIB file
    ###################################
    #mibCompile() {
    # TMPMIB=/tmp/mib.defs.xmservd
    # CATMIB=/tmp/mib.defs.out
    # xmservdMib $TMPMIB
    #
    # # echo "Compiling new MIB file"
    # mosy -o $CATMIB $TMPMIB
    # mibAppend $MIBDEFS $CATMIB
    # #cat $CATMIB >> $MIBDEFS
    #}


    ###################################
    # Name: snmpdStart
    # Purpose: start snmp daemons
    ###################################
    snmpdStart() {
    echo "Starting SNMP daemons"
    startsrc -s snmpd;startsrc -s dpid2;startsrc -s aixmibd;startsrc -s hostmibd;startsrc -s snmpmibd
    #wait
    }

    ###################################
    # Name: snmpdStop
    # Purpose: stop snmp daemons
    ###################################
    snmpdStop() {
    echo "Stopping SNMP daemons"
    stopsrc -s snmpd;stopsrc -s dpid2;stopsrc -s aixmibd;stopsrc -s hostmibd;stopsrc -s snmpmibd
    #wait
    }


    ###################################
    # Name: inetdConf
    # Purpose: snmpdv3.conf file
    ###################################
    inetdConfChange() {
    FILE=$1
    echo "Checking $FILE"
    if [ ! "$(grep "^xmquery" ${FILE})" ] ; then
    echo "Changing $FILE"
    echo 'xmquery dgram udp wait root /usr/bin/xmtopas xmtopas -p3' >> $FILE
    fi

    }


    ###################################
    # Name: initTab
    # Purpose: snmpdv3.conf file
    ###################################
    initTab() {

    if [ "$(lsitab xmdaily)" ] ; then
    echo "Remarking xmdaily in inittab"
    rmitab xmdaily
    fi

    if [ ! "$(lsitab topasrec)" ] ; then
    echo "Adding topas to inittab"
    mkitab "topasrec:2:once:/usr/bin/topas -R > /dev/null 2>&1 # Start topas recording"
    fi
    }


    ###################################
    # Name: snmpdPeers
    # Purpose: snmpdv3.conf file
    ###################################
    snmpdPeers() {
    FILE=$1
    echo "Checking $FILE"
    if [ ! "$(grep "^\"gated\"" ${FILE})" ] ; then
    echo "Adding gated line to $FILE"
    echo "\"gated\" 1.3.6.1.4.1.2.3.1.2.1.2 \"gated_password\"" >> $FILE
    fi
    if [ ! "$(grep "^\"dpid2\"" ${FILE})" ] ; then
    echo "Adding dpid2 line to $FILE"
    echo "\"dpid2\" 1.3.6.1.4.1.2.3.1.2.2.1.1.2 \"dpid_pw\"" >> $FILE
    fi
    if [ ! "$(grep "^\"muxatmd\"" ${FILE})" ] ; then
    echo "Adding muxatmd line to $FILE"
    echo "\"muxatmd\" 1.3.6.1.4.1.2.3.1.2.3.1.1 \"muxatmd_pw\"" >> $FILE
    fi
    if [ ! "$(grep "^\"xmservd\"" ${FILE})" ] ; then
    echo "Adding xmservd line to $FILE"
    echo "\"xmservd\" 1.3.6.1.4.1.2.3.1.2.1.3 \"xmservd_pw\"" >> $FILE
    fi
    }


    ###################################
    # Name: snmpdConf
    # Purpose: snmpdv3.conf file
    ###################################
    snmpdConf() {
    FILE=$1
    echo "Creating new $FILE"
    cat > $FILE << EOD
    # CONFIG GENERATED BY aix-snmp.sh

    VACM_GROUP group1 SNMPv1 public -

    VACM_VIEW defaultView internet - included -

    # exclude snmpv3 related MIBs from the default view
    VACM_VIEW defaultView snmpModules - excluded -
    VACM_VIEW defaultView 1.3.6.1.6.3.1.1.4 - included -
    VACM_VIEW defaultView 1.3.6.1.6.3.1.1.5 - included -

    # exclude aixmibd managed MIBs from the default view
    #VACM_VIEW defaultView 1.3.6.1.4.1.2.6.191 - excluded -
    VACM_VIEW defaultView 1.3.6.1.4.1.2.6.191 - included -

    VACM_ACCESS group1 - - noAuthNoPriv SNMPv1 defaultView - defaultView -

    NOTIFY notify1 traptag trap -

    TARGET_ADDRESS Target1 UDP 127.0.0.1 traptag trapparms1 - - -

    TARGET_PARAMETERS trapparms1 SNMPv1 SNMPv1 public noAuthNoPriv -

    COMMUNITY public public noAuthNoPriv 0.0.0.0 0.0.0.0 -

    DEFAULT_SECURITY no-access - -

    #trap public 172.20.4.132 fe

    logging file=/usr/tmp/snmpdv3.log enabled
    logging size=100000 level=0

    smux 1.3.6.1.4.1.2.3.1.2.1.2 gated_password # gated

    smux 1.3.6.1.4.1.2.3.1.2.3.1.1 muxatmd_password #muxatmd
    smux 1.3.6.1.4.1.2.3.1.2.1.3 xmservd_pw # xmservd

    smux 1.3.6.1.4.1.2.3.1.2.2.1.1.2 dpid_password #dpid


    EOD
    }


    Main;
